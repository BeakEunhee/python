def make_Kmeans(cluster, X, arr):
    kmeans = KMeans(n_clusters = cluster)
    kmeans.fit(X)
    mglearn.discrete_scatter(X[:, 0]                    ## 원래 값을 (군집 색을 넣어서 출력)
                           , X[:, 1]
                           , kmeans.labels_
                           , markers = 'o')
    cen = kmeans.cluster_centers_                       ## 클러스터 중심을 표시
    mglearn.discrete_scatter(cen[:, 0]
                           , cen[:, 1]
                           , arr
                           , markers = '^')
    
    return kmeans
